** manateed 的打包yaml 文件样例.

#+BEGIN_EXAMPLE
root@x200:/home/yuting/p/rangestack/manateed# cat index.yaml 

---
default:
  name: manateed
  requires:
    - perl
    - fixperl
    - daemontools076
    - iproute
    - ucspi-tcp
  summary: Allow remote execution of commands for cluster administration.
  version: '2.4.9'
  release: 1
  filetransforms:
    /usr/local/manateed/VERSION:
      - version: %version%
      - release: %release%
#+END_EXAMPLE
